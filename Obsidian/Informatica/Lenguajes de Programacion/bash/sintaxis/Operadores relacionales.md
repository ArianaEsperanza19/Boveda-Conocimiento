```bash
#!/bin/bash

numA=4

numB=2

# Esta es la sintaxis para hacer operaciones relacionales

echo $((numA < numB)) # menor que

echo $((numA > numB)) # mayor que

echo $((numA >= numB)) # mayor o igual

echo $((numA <= numB)) # menor o igual

echo $((numA == numB)) # igual

echo $((numA != numB)) # diferente

# El resultado que devolvera sera 0 para falso y 1 para verdadero
```
